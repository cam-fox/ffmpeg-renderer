FROM python:3.11-slim# instalar ffmpeg y dependenciasRUN apt-get update && apt-get install -y ffmpeg gcc libpq-dev && rm -rf /var/lib/apt/lists/*WORKDIR /appCOPY requirements.txt .RUN pip install --no-cache-dir -r requirements.txtCOPY . .ENV PORT=3000CMD ["uvicorn", "main:app", "--host", "0.0.0.0", "--port", "3000"]
